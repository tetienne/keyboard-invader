    **1. Create `wrangler.toml`** (per D-01, RESEARCH.md Pattern 2 -- Workers Static Assets, NOT legacy Pages):
    ```toml

    compatibility_date = "2026-03-27"

    [assets]
    directory = "./dist"
    not_found_handling = "single-page-application"
    ```
    IMPORTANT: No `main` field and no `binding` field -- this is assets-only mode. Do NOT use `wrangler pages deploy` (deprecated). Use `wrangler deploy`.

    **2. Create `.github/workflows/ci.yml`** (per D-02, RESEARCH.md GitHub Actions example):
    ```yaml
    name: CI/CD

    on:
      push:
        branches: [main]
      pull_request:
        branches: [main]

    jobs:
      check:
        runs-on: ubuntu-latest
        steps:
          - uses: actions/checkout@v4

          - uses: jdx/mise-action@v2

          - name: Install dependencies
            run: pnpm install --frozen-lockfile

          - name: Lint
            run: pnpm exec eslint .

          - name: Typecheck
            run: pnpm exec tsc --noEmit

          - name: Test
            run: pnpm exec vitest run

          - name: Build
            run: pnpm run build

      deploy:
        needs: check
        if: github.ref == 'refs/heads/main' && github.event_name == 'push'
        runs-on: ubuntu-latest
        permissions:
          contents: read
          deployments: write
        steps:
          - uses: actions/checkout@v4

          - uses: jdx/mise-action@v2

          - name: Install dependencies
            run: pnpm install --frozen-lockfile

          - name: Build
            run: pnpm run build

          - name: Deploy to Cloudflare Workers
            uses: cloudflare/wrangler-action@v3
            with:
              apiToken: ${{ secrets.CLOUDFLARE_API_TOKEN }}
              accountId: ${{ secrets.CLOUDFLARE_ACCOUNT_ID }}
    ```
    - Uses `jdx/mise-action@v2` to install Node + pnpm via mise (per D-05)
    - Deploy job runs only on push to main (per D-02)
    - Uses `cloudflare/wrangler-action@v3` (NOT pages-action, per RESEARCH.md Pitfall 1)
