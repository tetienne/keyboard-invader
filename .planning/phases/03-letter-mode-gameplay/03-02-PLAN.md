export type StateName = 'boot' | 'menu' | 'playing' | 'paused' | 'gameover'

export const TRANSITIONS: Record<StateName, readonly StateName[]> = {
  boot: ['menu'],
  menu: ['playing'],
  playing: ['paused', 'menu', 'gameover'],
  paused: ['playing', 'menu'],
  gameover: ['menu', 'playing'],
} as const

export interface SessionResult {
  readonly hits: number
  readonly misses: number
  readonly total: number
}

export interface GameContext {
  readonly app: Application
  readonly gameRoot: Container
  transitionTo(state: StateName): void
  getInputBuffer(): string[]
  acquirePoolItem(): { item: unknown; index: number }
  releasePoolItem(index: number): void
  readonly poolActiveCount: number
  readonly poolTotalCount: number
  readonly currentStateName: StateName
  setSessionResult(result: SessionResult): void
  getSessionResult(): SessionResult | null
}

    update(ctx, dt):
    - **Spawn logic:** Increment spawnTimer by dt. While spawnTimer >= SPAWN_INTERVAL_MS and totalSpawned < SESSION_LENGTH:
      - Subtract SPAWN_INTERVAL_MS from spawnTimer
      - Get available letters via `getAvailableLetters(this.totalSpawned, this.SESSION_LENGTH)` (per D-04)
      - Pick random letter from available pool
      - Acquire pool item, cast to BitmapText
      - Reset all properties (Pitfall 2): `bt.text = letter.toUpperCase()` (display uppercase for kids), pick random color index `const colorIdx = Math.floor(Math.random() * LETTER_COLORS.length)`, set `bt.tint = LETTER_COLORS[colorIdx] ?? 0xffffff`, `bt.scale.set(1)`, `bt.alpha = 1`, `bt.anchor.set(0.5)`, `bt.x = 80 + Math.random() * (BASE_WIDTH - 160)`, `bt.y = -40`, `bt.visible = true`
      - Add to gameRoot, push LetterEntity to activeEntities with `letter` (lowercase), `baseX = bt.x`, `tween: null`, `markedForRemoval: false`, `originalTint = bt.tint`
      - Increment totalSpawned

    - **Fall logic:** For each entity in activeEntities where tween is null and markedForRemoval is false: `entity.text.y += FALL_SPEED * (dt / 1000)`

    - **Input processing (D-13, D-14, D-15):** Get input buffer via ctx.getInputBuffer(). For each key:
      - Call `findLowestMatch(activeEntities, key)`
      - If match found: increment hits, set `match.tween = createHitTween()`, set `match.markedForRemoval = true`
      - If no match: increment misses. Call `findLowestEntity(activeEntities)` -- if found and its tween is null, set `lowest.tween = createMissTween()` (D-08: red flash on lowest letter). Miss tween does NOT mark for removal.

    - **Update score display:** `if (this.scoreText) this.scoreText.text = 'Score: ' + String(this.hits)` (per D-10)

    - **Tween updates:** For each entity in activeEntities: if entity.tween is not null, call `updateTween(entity, dt)`. If returns true (complete):
      - If tween type was 'hit' or 'bottom': entity already marked for removal (was marked when tween started)
      - If tween type was 'miss': reset entity visual -- `entity.text.tint = entity.originalTint`, `entity.text.x = entity.baseX`, set `entity.tween = null` (entity stays active, just visual feedback done)

    - **Bottom detection (D-09):** For each entity where tween is null and not markedForRemoval: if `entity.text.y > BASE_HEIGHT + 40`, set `entity.tween = createBottomTween()`, set `entity.markedForRemoval = true`.

    - **Cleanup pass (Pitfall 3):** Iterate activeEntities in reverse. For each entity where markedForRemoval is true AND tween is null (tween completed and was nulled, or never had one): `entity.text.visible = false`, `ctx.gameRoot.removeChild(entity.text)`, `ctx.releasePoolItem(entity.poolIndex)`, splice from array. For hit/bottom tweens, also null the tween after updateTween returns true so that next cleanup pass removes it.

    - **Session end check (D-11):** If totalSpawned >= SESSION_LENGTH and activeEntities.length === 0: call `ctx.setSessionResult({ hits: this.hits, misses: this.misses, total: this.SESSION_LENGTH })`, then `ctx.transitionTo('gameover')`.

    **Part C -- Add GameOverState to src/game/states.ts:**

    Per D-12 and D-16:
    ```typescript
    export class GameOverState implements GameState {
      private container: Container | null = null

      enter(ctx: GameContext): void {
        const result = ctx.getSessionResult()
        this.container = new Container()

        // Title -- encouraging "Bravo !" text
        const title = new BitmapText({
          text: 'Bravo !',
          style: { fontFamily: 'GameFont', fontSize: 48 },
        })
        title.anchor.set(0.5)
        title.x = BASE_WIDTH / 2
        title.y = BASE_HEIGHT * 0.2

        // Stats display
        const hits = result?.hits ?? 0
        const misses = result?.misses ?? 0
        const total = result?.total ?? 0
        const accuracy = total > 0 ? Math.round((hits / total) * 100) : 0

        const statsText = new BitmapText({
          text: `${String(hits)} touches  |  ${String(misses)} rates  |  ${String(accuracy)}% precision`,
          style: { fontFamily: 'GameFont', fontSize: 20 },
        })
        statsText.anchor.set(0.5)
        statsText.x = BASE_WIDTH / 2
        statsText.y = BASE_HEIGHT * 0.45

        // "Rejouer" button (D-12)
        const replayBtn = new BitmapText({
          text: 'Rejouer',
          style: { fontFamily: 'GameFont', fontSize: 28 },
        })
        replayBtn.anchor.set(0.5)
        replayBtn.x = BASE_WIDTH / 2
        replayBtn.y = BASE_HEIGHT * 0.65
        replayBtn.eventMode = 'static'
        replayBtn.cursor = 'pointer'
        replayBtn.on('pointerover', () => { replayBtn.scale.set(1.1) })
        replayBtn.on('pointerout', () => { replayBtn.scale.set(1.0) })
        replayBtn.on('pointertap', () => { ctx.transitionTo('playing') })

        // "Menu" button
        const menuBtn = new BitmapText({
          text: 'Menu',
          style: { fontFamily: 'GameFont', fontSize: 20 },
        })
        menuBtn.anchor.set(0.5)
        menuBtn.x = BASE_WIDTH / 2
        menuBtn.y = BASE_HEIGHT * 0.78
        menuBtn.eventMode = 'static'
        menuBtn.cursor = 'pointer'
        menuBtn.on('pointerover', () => { menuBtn.scale.set(1.1) })
        menuBtn.on('pointerout', () => { menuBtn.scale.set(1.0) })
        menuBtn.on('pointertap', () => { ctx.transitionTo('menu') })

        this.container.addChild(title, statsText, replayBtn, menuBtn)
        ctx.gameRoot.addChild(this.container)
      }

      exit(ctx: GameContext): void {
        if (this.container) {
          ctx.gameRoot.removeChild(this.container)
          this.container.destroy({ children: true })
          this.container = null
        }
      }

      update(): void { /* static screen */ }
      render(): void { /* no-op */ }
    }
    ```
