export function getLevelTitle(level: number, locale: string): string {
  // Find highest level threshold <= player level
  const thresholds = Object.keys(LEVEL_TITLES).map(Number).sort((a, b) => b - a)

      const titles = LEVEL_TITLES[threshold]!
      return locale === 'en' ? titles.en : titles.fr

**Part A: Create SVG art assets**

All SVGs must follow these constraints per RESEARCH.md:
- Under 2KB each
- Simple paths, no gradients, no filters, no text elements
- Explicit width/height attributes on SVG root

Create directory `public/assets/aliens/` and `public/assets/avatars/`.

**Letter-mode aliens (6 files, 64x64 viewBox):**
Design round, blobby cute creatures with big eyes and soft shapes (per D-02). Each alien is visually distinct but shares the same style family. No sharp edges. Think Slime Rancher / Cut the Rope aesthetic.

- `alien-01.svg`: Round blob with two big eyes, small smile. Simple circle body.
- `alien-02.svg`: Oval blob with one big eye (cyclops), happy expression.
- `alien-03.svg`: Triangular-ish blob with two eyes, antenna on top.
- `alien-04.svg`: Wide flat blob with two eyes, small arms/tentacles on sides.
- `alien-05.svg`: Tall blob with three small eyes stacked vertically, goofy smile.
- `alien-06.svg`: Squat round blob with big cheeks, two eyes, tiny horns.

Each alien body should be a solid fill color (will be tinted by PixiJS at runtime, so use white or light gray as base). Eyes should be simple black circles with white highlights.

**Word-mode aliens (2 files, 128x128 viewBox per D-07):**
Bigger, visually distinct from letter aliens. Multi-eyed or multi-armed.

- `word-alien-01.svg`: Larger round blob with 4 eyes, two small arms. 128x128 viewBox.
- `word-alien-02.svg`: Larger oval blob with 3 eyes in a row, four tentacles. 128x128 viewBox.

**Part B: Rewrite BootState to async-load assets + install Fredoka BitmapFont**

Modify `BootState.enter()` in `src/game/states.ts`. The current implementation is synchronous and installs Arial BitmapFont. Replace with:

1. Keep `enter()` signature as `void` (per RESEARCH.md Pitfall 5 recommendation: don't change GameState interface).
2. Inside enter(), start an async IIFE that:
   a. Imports `'@fontsource/fredoka/400.css'` and `'@fontsource/fredoka/700.css'` (these are top-level static imports added at file top)
   b. Awaits `document.fonts.load('400 80px Fredoka')`
   c. Awaits `document.fonts.load('700 48px Fredoka')`
   d. Installs BitmapFont with name 'GameFont', fontFamily 'Fredoka', fontSize 80, fill '#ffffff', resolution 2 (fixes D-22 blurriness)
   e. Installs a second BitmapFont with name 'GameFontBold', fontFamily 'Fredoka', weight '700', fontSize 48, fill '#ffffff', resolution 2
   f. Loads all SVG textures via `Assets.load()` from the paths in theme.ts (ALIEN_TEXTURES_PATHS, WORD_ALIEN_TEXTURE_PATHS, SPACESHIP_PATH, STAR_PARTICLE_PATH, and all AVATAR_SVG_PATHS values)
   g. After all loads complete, calls `ctx.transitionTo('profiles')`
   h. Wraps everything in try/catch. On error, log to console.error.

3. Add static imports at the top of states.ts:
```typescript
import '@fontsource/fredoka/400.css'
import '@fontsource/fredoka/700.css'
```

4. Also update the `app.init()` background color in `src/game/game.ts` from `'#1a1a2e'` to `'#1a1a3e'` to match the new SPACE_PALETTE.background.

**Important:** The `BitmapFont.install()` call must use `chars: BitmapFont.ASCII` to pre-generate the character set. Pass `resolution: 2` to fix blurriness (D-22).

**Important:** Do NOT change the `GameState` interface signature. BootState.enter() stays as `void` return type. The async loading happens inside a fire-and-forget async IIFE.
