**Per D-12 and D-13 — Create `tests/i18n/keys-used.test.ts`** with the following exact content:

```typescript
import { readFileSync, readdirSync, statSync } from 'node:fs'
import { resolve, join } from 'node:path'
import { describe, it, expect } from 'vitest'

const rootDir = resolve(import.meta.dirname, '..', '..')
const srcDir = resolve(rootDir, 'src')
const enPath = resolve(srcDir, 'shared', 'i18n', 'en.json')
const frPath = resolve(srcDir, 'shared', 'i18n', 'fr.json')

const EXCLUDED_FILES = new Set<string>([enPath, frPath])
const SCANNABLE_EXTENSIONS = new Set<string>(['.ts', '.tsx', '.js', '.json'])

function collectFiles(dir: string, acc: string[]): void {
  for (const entry of readdirSync(dir)) {
    const full = join(dir, entry)
    const st = statSync(full)
    if (st.isDirectory()) {
      collectFiles(full, acc)
    } else if (st.isFile()) {
      if (EXCLUDED_FILES.has(full)) continue
      const dot = full.lastIndexOf('.')
      if (dot === -1) continue
      const ext = full.slice(dot)
      if (SCANNABLE_EXTENSIONS.has(ext)) acc.push(full)
    }
  }
}

describe('i18n keys: every key in en.json is referenced somewhere in src/', () => {
  it('finds at least one literal occurrence per key (no orphans)', () => {
    const enRaw = readFileSync(enPath, 'utf-8')
    const enKeys = Object.keys(JSON.parse(enRaw) as Record<string, string>)

    const files: string[] = []
    collectFiles(srcDir, files)

    const fileContents = files.map((f) => readFileSync(f, 'utf-8'))

    const orphans: string[] = []
    for (const key of enKeys) {
      const found = fileContents.some((c) => c.includes(key))
      if (!found) orphans.push(key)
    }

    expect(orphans, `Orphan i18n keys (declared in en.json but never referenced in src/): ${orphans.join(', ')}`).toEqual([])
  })
})
```

**Notes for the executor (do not include as code comments in the file):**

- `import.meta.dirname` resolves to `tests/i18n` at runtime; `..` x 2 climbs to repo root. (Same pattern as tests/smoke.test.ts which uses `resolve(import.meta.dirname, '..')`.)
- The walk is dependency-free: `node:fs` `readdirSync` + `statSync` only. No ripgrep, no glob library.
- `EXCLUDED_FILES` is a Set of absolute paths covering both en.json and fr.json so a key string in those JSONs cannot satisfy itself. fr.json is excluded too because it mirrors en's keys and would create a false positive for any orphan key still present in fr.
- `SCANNABLE_EXTENSIONS` covers .ts, .tsx, .js, .json so keys used in both code (`t('profiles.title')`) and other JSON config files (if any) match.
- The assertion uses `expect([...]).toEqual([])` so the failure message lists the exact orphan keys — easy to fix.
- The test currently scans every src/ file each run. With ~50 keys and ~50 src files this is well under 100ms — fine for the test budget.

After Task 1's deletions, every remaining en.json key SHOULD have at least one match in src/. If this test fails on first run, that surfaces additional orphans — per D-16, capture them as deferred for a follow-up cleanup, do NOT delete them in this phase.
